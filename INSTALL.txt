#!/usr/bin/bash

echo " 1.)Mobile or 2.)PC : "

read answer

if [$answer == 1]; then
#Maintain and install requirements for Termux
  termuxinst() {
    apt update && apt upgrade -y
    pkg install golang -y
    GO111MODULE=on go get -v github.com/projectdiscovery/subfinder/v2/cmd/subfinder
    GO111MODULE=on go get -u github.com/sw33tLie/bbscope
    cd /data/data/com.termux/files/home/go/bin
    mv bbscope subfinder /data/data/com.termux/files/usr/bin
  }
  termuxinst
else
  apt update && apt upgrade -y
  apt install subfinder golang -y
  GO111MODULE=on go get -u github.com/sw33tLie/bbscope
  cd go/bin
  mv bbscope /usr/bin
fi
